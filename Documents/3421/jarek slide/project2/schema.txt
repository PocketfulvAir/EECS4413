CREATE TABLE Album(
    title       varchar(50)  notnull, 
    year        INT, 
    PRIMARY KEY(title)   
    ));  
    
    
CREATE TABLE musician(
    name        varchar(50) not null,      
    hometown    varchar(50) not null,
    instruments varchar(50) not null   
    PRIMARY KEY(name)
);

CREATE TABLE song(
    title       varchar(50) not null,
    genre       varchar(50) not null,
    PRIMARY KEY(title)
);

CREATE TABLE recording(
    title       varchar(50) not null,
    studio      varchar(50) not null,
    number      INT, 
    PRIMARY KEY (title) references Album,
    FOREIGN KEY (title) references song  
);
    
CREATE TABLE recording_comapany(
    title       varchar(50) not null,
    address     varchar(50) not null,
    title       varchar(50) not null, 
    homepage    varchar(50) not null,
    PRIMARY KEY(title)
    FOREIGN KEY(title) refernces Album
);

CREATE TABLE chart(
    name        varchar(20) not null,
    PRIMARY KEY(name)
);

CREATE TABLE charts(
    name  varchar(50) not null, 
    startday DATE not null, 
    rank INT, 
    PRIMARY KEY(name) refernces song,
    FOREIGN KEY(startday) refernces week
);

CREATE TABLE week (
    startday DATE not null,
    PRIMARY KEY(startday)
);

Create table Orchestra(
    name varchar(30) not null,
    Size INT,
    Conductor varchar(30),
    title varchar(15),
    Primary Key(name), 
    Foreign key(title) refernces Album
);
